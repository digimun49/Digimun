rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Signal counter - authenticated users can only increment by 1
    match /stats/signalCount {
      allow read: if true;
      allow write: if request.auth != null
        && request.resource.data.count == resource.data.count + 1;
    }
    
    // Support tickets
    match /tickets/{ticketId} {
      allow create: if true;
      allow read: if request.auth != null
        && (
          resource.data.email == request.auth.token.email
          || request.auth.token.email == "muneebg249@gmail.com"
        );
      allow update: if request.auth != null
        && (
          resource.data.email == request.auth.token.email
          || request.auth.token.email == "muneebg249@gmail.com"
        );
      allow delete: if request.auth != null
        && request.auth.token.email == "muneebg249@gmail.com";
    }
    
    // Reviews - force pending status on create
    match /reviews/{reviewId} {
      allow create: if request.resource.data.status == "pending";
      allow read: if resource.data.status == "approved";
      allow read, update, delete:
        if request.auth != null
        && request.auth.token.email == "muneebg249@gmail.com";
    }
    
    // User profiles
    match /users/{userEmail} {
      // Users can CREATE their own document with safe initial values only
      allow create: if request.auth != null
        && request.auth.token.email.lower() == userEmail.lower()
        && request.resource.data.status == "approved"
        && request.resource.data.paymentStatus == "pending"
        && request.resource.data.quotexStatus == "pending"
        && request.resource.data.digimaxStatus == "pending"
        && request.resource.data.recoveryRequest == "pending"
        && request.resource.data.DigimunXAdv == "pending";
      
      // Users can only read their own profile
      allow read: if request.auth != null
        && request.auth.token.email.lower() == userEmail.lower();
      
      // Users can only update safe fields (contact info, NOT status fields)
      allow update: if request.auth != null
        && request.auth.token.email.lower() == userEmail.lower()
        && !request.resource.data.diff(resource.data).affectedKeys()
            .hasAny(['paymentStatus', 'quotexStatus', 'digimaxStatus', 'recoveryRequest', 'status', 'approvedAt', 'lockedAt', 'accountStatus', 'DigimunXAdv']);
      
      // Admin has full access
      allow read, write:
        if request.auth != null
        && request.auth.token.email == "muneebg249@gmail.com";
    }
    
    // Page visit tracking - anyone can create, only admin can read
    match /pageVisits/{visitId} {
      allow create: if true;
      allow read: if request.auth != null
        && request.auth.token.email == "muneebg249@gmail.com";
    }
    
    // Active visitors tracking - anyone can create/update/delete their own, admin can read all
    match /activeVisitors/{visitorId} {
      allow create, update, delete: if true;
      allow read: if request.auth != null
        && request.auth.token.email == "muneebg249@gmail.com";
    }
  }
}
