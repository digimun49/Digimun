    <script>
        // ⚠️ IMPORTANT: Replace with your actual Replit deployment URL
        const API_BASE = 'https://YOUR-REPLIT-URL.replit.app';
        
        // Fetch Stats
        async function fetchStats() {
            try {
                const response = await fetch(`${API_BASE}/api/stats`);
                const data = await response.json();
                
                if (data.success) {
                    const stats = data.stats;
                    
                    // Overall Stats
                    document.getElementById('totalSignals').textContent = stats.overall.total_signals;
                    document.getElementById('winRate').textContent = stats.overall.win_rate + '%';
                    
                    // Today Stats
                    document.getElementById('todaySignals').textContent = stats.today.total_signals;
                    document.getElementById('todayWinRate').textContent = stats.today.win_rate + '%';
                    
                    // Top Pairs
                    const topPairsContainer = document.getElementById('topPairs');
                    topPairsContainer.innerHTML = '<strong style="margin-right: 10px;">🏆 Top Pairs:</strong>';
                    stats.top_pairs.forEach(pair => {
                        topPairsContainer.innerHTML += `
                            <div class="pair-chip">${pair.pair} (${pair.win_rate}% win)</div>
                        `;
                    });
                }
            } catch (error) {
                console.error('Error fetching stats:', error);
            }
        }
        // Fetch Signals
        async function fetchSignals() {
            try {
                const response = await fetch(`${API_BASE}/api/signals?limit=50`);
                const data = await response.json();
                
                if (data.success) {
                    const tbody = document.getElementById('signalsBody');
                    tbody.innerHTML = '';
                    
                    data.signals.forEach(signal => {
                        // Direction badge
                        const directionClass = signal.direction === 'CALL' ? 'badge-call' : 'badge-put';
                        const directionEmoji = signal.direction === 'CALL' ? '📈' : '📉';
                        
                        // Result class
                        let resultClass = 'result-pending';
                        let resultText = '⏳ Pending';
                        if (signal.result === 'WIN') {
                            resultClass = 'result-win';
                            resultText = '✅ WIN';
                        } else if (signal.result === 'LOSS') {
                            resultClass = 'result-loss';
                            resultText = '❌ LOSS';
                        }
                        
                        // Confidence color
                        let confClass = 'confidence-high';
                        if (signal.confidence < 70) confClass = 'confidence-low';
                        else if (signal.confidence < 85) confClass = 'confidence-medium';
                        
                        tbody.innerHTML += `
                            <tr>
                                <td><strong>${signal.id}</strong></td>
                                <td>${signal.time}</td>
                                <td><strong>${signal.pair}</strong></td>
                                <td><span class="badge ${directionClass}">${directionEmoji} ${signal.direction}</span></td>
                                <td>
                                    <div style="display: flex; align-items: center; gap: 10px;">
                                        <div class="confidence-bar">
                                            <div class="confidence-fill ${confClass}" style="width: ${signal.confidence}%"></div>
                                        </div>
                                        <span>${signal.confidence}%</span>
                                    </div>
                                </td>
                                <td class="${resultClass}">${resultText}</td>
                            </tr>
                        `;
                    });
                    
                    // Update timestamp
                    document.getElementById('lastUpdate').textContent = 'Updated: ' + new Date().toLocaleTimeString();
                }
            } catch (error) {
                console.error('Error fetching signals:', error);
                document.getElementById('signalsBody').innerHTML = '<tr><td colspan="6" style="text-align: center; color: #ff4444;">Error loading signals. Please refresh.</td></tr>';
            }
        }
        // Initial load
        fetchStats();
        fetchSignals();
        // Auto-refresh every 30 seconds
        setInterval(() => {
            fetchStats();
            fetchSignals();
        }, 30000);
    </script>
