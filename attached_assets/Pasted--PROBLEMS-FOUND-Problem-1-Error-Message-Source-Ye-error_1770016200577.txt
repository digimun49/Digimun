Fix 2: Check API Response Has signal_id
async function loadSignals() {
    const response = await fetch(`${SIGNALS_API}/api/signals?limit=500`);
    const data = await response.json();
    
    console.log('API Response:', data.signals[0]); // Debug first signal
    
    // Check if signal_id exists
    if (data.signals[0] && !data.signals[0].signal_id) {
        console.error('WARNING: signal_id missing from API response!');
    }
}
Fix 3: Use New POST Endpoint (EASIER)
async function deleteSignalById(signalId) {
    const response = await fetch(`${SIGNALS_API}/api/signals/action`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            signal_id: signalId,
            action: 'delete'
        })
    });
    const data = await response.json();
    console.log('Delete response:', data);
}
